203  template<typename T>
204  boost::shared_ptr<T> FindParentType( bool a_bEnumProxies = false ) const
205  {
206  SP spParent = GetParent( a_bEnumProxies );
207  while (spParent)
208  {
209  boost::shared_ptr<T> spCasted = DynamicCast<T>(spParent);
210  if (spCasted)
211  return spCasted;
212  spParent = spParent->GetParent( a_bEnumProxies );
213  }
214 
215  return boost::shared_ptr<T>();
216  }

219  template<typename T>
220  bool FindChildrenType( std::vector< boost::shared_ptr<T> > & a_Children, bool a_bRecursive = true ) const
221  {
222  for(size_t i=0;i<m_Children.size();++i)
223  {
224  const SP & spChild = m_Children[i];
225  if ( a_bRecursive )
226  spChild->FindChildrenType<T>( a_Children, true );
227 
228  boost::shared_ptr<T> spCasted = DynamicCast<T>( spChild );
229  if ( spCasted )
230  a_Children.push_back( spCasted );
231  }
232 
233  return a_Children.size() > 0;
234  }

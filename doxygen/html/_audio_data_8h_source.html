87  static bool ParseAudioFormat( const std::string & a_Format,
88  unsigned int & a_Freq, unsigned int & a_BPS, unsigned int & a_Channels )
89  {
90  std::vector<std::string> parts;
91  StringUtil::Split( a_Format, ";", parts );
92 
93  if ( parts.size() > 0 && StringUtil::Compare( parts[0], "audio/L16", true ) == 0 )
94  {
95  a_BPS = 16;
96  a_Freq = 16000;
97  a_Channels = 1;
98 
99  for(size_t i=1;i<parts.size();++i)
100  {
101  std::vector<std::string> kv;
102  StringUtil::Split( parts[i], "=", kv );
103  if ( kv.size() == 2 )
104  {
105  if ( StringUtil::Compare( kv[0], "rate", true ) == 0 )
106  a_Freq = atoi( kv[1].c_str() );
107  else if ( StringUtil::Compare( kv[0], "channels", true ) == 0 )
108  a_Channels = atoi( kv[1].c_str() );
109  }
110  }
111 
112  return true;
113  }
114 
115  return false;
116  }

43  // Analyze the samples in <code>buffer</code>. This is a cumulative
44  // process, so you must call this function every frame.
45  virtual bool Detect( std::vector<float> & a_Samples, size_t a_nOffset )
46  {
47  if ( a_nOffset >= (int)a_Samples.size() )
48  throw WatsonException( "Buffer overflow" );
49  if ( (((int)a_Samples.size()) - a_nOffset) < m_nTimeSize )
50  throw WatsonException( "Buffer underflow" );
51 
52  // compute the energy level
53  float level = 0;
54  for (int i = 0; i < m_nTimeSize; i++)
55  level += (a_Samples[i + a_nOffset] * a_Samples[i + a_nOffset]);
56 
57  level /= m_nTimeSize;
58  level = (float) sqrt(level);
59  float instant = level * 100;
60  // compute the average local energy
61  float E = GetAverage(m_eBuffer);
62  // compute the variance of the energies in eBuffer
63  float V = GetVariance(m_eBuffer, E);
64  // compute C using a linear digression of C with V
65  float C = (-0.0025714f * V) + 1.5142857f;
66  // filter negaive values
67  float diff = (float)MAX(instant - C * E, 0);
68  // find the average of only the positive values in dBuffer
69  float dAvg = GetSpecAverage(m_dBuffer);
70  // filter negative values
71  float diff2 = (float)MAX(diff - dAvg, 0);
72 
73  // report false if it's been less than 'sensitivity'
74  // milliseconds since the last true value
75  double now = Time().GetEpochTime();
76  if ( (now - m_Timer) < m_Sensitivity)
77  {
78  m_bIsOnset = false;
79  }
80  // if we've made it this far then we're allowed to set a new
81  // value, so set it true if it deserves to be, restart the timer
82  else if (diff2 > 0 && instant > 2)
83  {
84  m_bIsOnset = true;
85  m_Timer = now;
86  }
87  // OMG it wasn't true!
88  else
89  {
90  m_bIsOnset = false;
91  }
92  m_eBuffer[m_nInsertAt] = instant;
93  m_dBuffer[m_nInsertAt] = diff;
94 
95  if (++m_nInsertAt == m_eBuffer.size())
96  m_nInsertAt = 0;
97 
98  return m_bIsOnset;
99  }
100

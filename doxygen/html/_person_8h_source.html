111  float GetFaceDistance( const std::vector<float> & a_Location ) const
112  {
113  float distance = 0.0f;
114  for(size_t i=0;i<a_Location.size();++i)
115  {
116  if ( i >= m_FaceLocation.size() )
117  break;
118 
119  float x = a_Location[i] - m_FaceLocation[i];
120  distance += (x * x);
121  }
122 
123  return sqrt( distance );
124  }

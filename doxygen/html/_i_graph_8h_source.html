131  static bool SortEdges( const EdgeSP & a1, const EdgeSP & a2 )
132  {

134  }
135 
136  bool FindOutEdges( const char * a_pEdgeLabel,
137  EdgeList & a_Edges )
138  {
139  for( size_t i=0;i<m_OutEdges.size(); ++i)
140  {
141  const IEdge::SP & spEdge = m_OutEdges[i];
142  if ( (*spEdge)["label"].asString().compare( a_pEdgeLabel ) != 0 )
143  continue;
144  a_Edges.push_back( spEdge );
145  }
146 
147  std::sort( a_Edges.begin(), a_Edges.end(), SortEdges );
148  return a_Edges.size() > 0;
149  }
150 
151  bool FindInEdges( const char * a_pEdgeLabel,
152  EdgeList & a_Edges )
153  {
154  for( size_t i=0;i<m_InEdges.size(); ++i)
155  {
156  const IEdge::SP & spEdge = m_InEdges[i];
157  if ( (*spEdge)["label"].asString().compare( a_pEdgeLabel ) != 0 )
158  continue;
159  a_Edges.push_back( spEdge );
160  }
161 
162  std::sort( a_Edges.begin(), a_Edges.end(), SortEdges );
163  return a_Edges.size() > 0;
164  }

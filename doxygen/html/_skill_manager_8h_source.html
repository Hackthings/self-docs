164 inline bool SkillManager::AbortActiveSkills()
165 {
166  for( SkillSet::iterator iSkill = m_ActiveSkills.begin();
167  iSkill != m_ActiveSkills.end(); iSkill = m_ActiveSkills.begin() )
168  {
169  if (! (*iSkill)->Abort() )
170  return false; // failed to abort this skill
171 
172  delete *iSkill;
173  }
174  return true;
175 }

177 inline bool SkillManager::AbortActiveSkill(const std::string & a_Skill)
178 {
179  for( SkillSet::iterator iSkill = m_ActiveSkills.begin(); iSkill != m_ActiveSkills.end(); ++iSkill )
180  {
181  if(!(*iSkill)->GetSkillGUID().compare(a_Skill))
182  {
183  if(! (*iSkill)->Abort() )
184  return false;
185 
186  delete *iSkill;
187  return true;
188  }
189  }
190  return false;
191 }

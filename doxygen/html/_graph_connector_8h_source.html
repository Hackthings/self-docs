233  template<typename T>
234  struct EventRequest : public IEventRequest
235  {
236  EventRequest( GraphConnector * a_pConnector,
237  EventId a_nEventId,
238  Delegate<T> a_Callback,
239  T a_CallbackArg,
240  float a_fTimeout ) :
241  m_pConnector(a_pConnector),
242  m_nEventId( a_nEventId ),
243  m_Callback( a_Callback ),
244  m_CallbackArg( a_CallbackArg )
245  {
246  m_pConnector->m_EventMap[ a_nEventId ] = IEventRequest::SP( this );
247  if ( TimerPool::Instance() != NULL )
248  {
249  m_spTimer = TimerPool::Instance()->StartTimer(
250  VOID_DELEGATE( EventRequest, OnEventTimeout, this ), a_fTimeout, true, false );
251  }
252  }
253 
254  GraphConnector * m_pConnector;
255  EventId m_nEventId;
256  Delegate<T> m_Callback;
257  T m_CallbackArg;
258  TimerPool::ITimer::SP m_spTimer;
259 
260  virtual void OnEventDone()
261  {
262  m_spTimer.reset();
263  if ( m_Callback.IsValid() )
264  m_Callback( m_CallbackArg );
265  m_pConnector->m_EventMap.erase( m_nEventId );
266  }
267  virtual void OnEventTimeout()
268  {
269  Log::Warning( "GraphConnector", "Timeout of event %u", m_nEventId );
270  OnEventDone();
271  }
272  };

324 template<typename T>
325 inline GraphConnector::EventId GraphConnector::Send( const IGraphEvent & a_Event,
326  Delegate<T> a_Callback,
327  T a_CallbackArg,
328  float a_fTimeoout /*= 30.0f*/ )
329 {
330  IGraphEvent & ev = const_cast<IGraphEvent &>( a_Event );
331  ev.m_nEventId = m_nNextEventId++;
332  if (! Send( ev ) )
333  return 0;
334 
335  new EventRequest<T>( this, ev.m_nEventId, a_Callback, a_CallbackArg, a_fTimeoout );
336  return ev.m_nEventId;
337 }
